        self.short_term["results"].append(result)
    
    def remember(self, key: str, value: str):
        """–ó–∞–ø–æ–º–∏–Ω–∞–µ—Ç —Ñ–∞–∫—Ç –Ω–∞–¥–æ–ª–≥–æ"""

                zip(self.short_term["completed_steps"], self.short_term["results"])

        if self.long_term:
            context += "\n–ò–∑–≤–µ—Å—Ç–Ω—ã–µ —Ñ–∞–∫—Ç—ã:\n"
            for key, value in self.long_term.items():
                context += f"- {key}: {value}\n"
        
        return context
