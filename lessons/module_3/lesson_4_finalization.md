def parse_command(user_input: str) -> tuple:
    """–†–∞–∑–±–∏—Ä–∞–µ—Ç –∫–æ–º–∞–Ω–¥—É –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è"""
    parts = user_input.strip().split(maxsplit=1)
    command = parts[0].lower() if parts else ""
    args = parts[1] if len(parts) > 1 else ""
    return command, args

    """–°–æ–∑–¥–∞—ë—Ç —Ä–∞–∑–¥–µ–ª–∏—Ç–µ–ª—å"""
    return char * length

## 2. –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ –∏ –∑–∞–≥—Ä—É–∑–∫–∞ —Å–µ—Å—Å–∏–π

```python
"""
–ú–µ—Ç–æ–¥—ã –¥–ª—è —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏—è –∏ –∑–∞–≥—Ä—É–∑–∫–∏ —Å–µ—Å—Å–∏–π.

# –í –Ω–∞—á–∞–ª–µ —Ñ–∞–π–ª–∞ –¥–æ–±–∞–≤—å—Ç–µ:
SESSIONS_DIR = "sessions"


class SchoolAssistant:
    # ... —Å—É—â–µ—Å—Ç–≤—É—é—â–∏–π –∫–æ–¥ ...
    
    def save_session(self, name: str) -> bool:
        """
        –°–æ—Ö—Ä–∞–Ω—è–µ—Ç —Ç–µ–∫—É—â—É—é —Å–µ—Å—Å–∏—é –≤ JSON —Ñ–∞–π–ª.
        
        Args:

        Returns:
            True –µ—Å–ª–∏ —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ —É—Å–ø–µ—à–Ω–æ
        """
        # –°–æ–∑–¥–∞—ë–º –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏—é –µ—Å–ª–∏ –Ω–µ—Ç
        if not os.path.exists(SESSIONS_DIR):

        session_data = {

            return True
        except Exception as e:
            print(f"‚ùå –û—à–∏–±–∫–∞ —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏—è: {e}")

    def load_session(self, name: str) -> bool:
        """
        –ó–∞–≥—Ä—É–∂–∞–µ—Ç —Å–µ—Å—Å–∏—é –∏–∑ JSON —Ñ–∞–π–ª–∞.
        
        Args:
            name: –ò–º—è —Å–µ—Å—Å–∏–∏
            
        Returns:
            True –µ—Å–ª–∏ –∑–∞–≥—Ä—É–∑–∫–∞ —É—Å–ø–µ—à–Ω–∞
        """
        filename = os.path.join(SESSIONS_DIR, f"{name}.json")

        try:
            with open(filename, 'r', encoding='utf-8') as f:
                session_data = json.load(f)

            settings = session_data.get("settings", {})

            context_data = session_data.get("context", {})

            self._update_system_prompt()

            questions_data = session_data.get("current_questions", [])
            if questions_data and hasattr(self, 'current_questions'):
                from exam_quiz import Question
                self.current_questions = [

    def list_sessions(self) -> List[str]:
        """
        –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Å–ø–∏—Å–æ–∫ —Å–æ—Ö—Ä–∞–Ω—ë–Ω–Ω—ã—Ö —Å–µ—Å—Å–∏–π.
        
        Returns:
            –°–ø–∏—Å–æ–∫ –∏–º—ë–Ω —Å–µ—Å—Å–∏–π
        """
        if not os.path.exists(SESSIONS_DIR):
            return []
        
        sessions = []
        for filename in os.listdir(SESSIONS_DIR):
            if filename.endswith('.json'):
                name = filename[:-5]  # –£–±–∏—Ä–∞–µ–º .json
                
                # –ü—Ä–æ–±—É–µ–º –ø–æ–ª—É—á–∏—Ç—å –¥–∞—Ç—É —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏—è
                filepath = os.path.join(SESSIONS_DIR, filename)
                try:
                    with open(filepath, 'r', encoding='utf-8') as f:
                        data = json.load(f)
                        saved_at = data.get("saved_at", "")
                        if saved_at:
                            # –§–æ—Ä–º–∞—Ç–∏—Ä—É–µ–º –¥–∞—Ç—É
                            dt = datetime.fromisoformat(saved_at)
                            date_str = dt.strftime("%d.%m.%Y %H:%M")
                            sessions.append(f"{name} ({date_str})")
                        else:
                            sessions.append(name)
                except:
                    sessions.append(name)
        
        return sorted(sessions, reverse=True)
    
    def delete_session(self, name: str) -> bool:

        if os.path.exists(filename):
            try:
                os.remove(filename)
                print(f"üóëÔ∏è –°–µ—Å—Å–∏—è —É–¥–∞–ª–µ–Ω–∞: {name}")

            except Exception as e:
                print(f"‚ùå –û—à–∏–±–∫–∞ —É–¥–∞–ª–µ–Ω–∏—è: {e}")
                return False
        else:
            print(f"‚ùå –°–µ—Å—Å–∏—è –Ω–µ –Ω–∞–π–¥–µ–Ω–∞: {name}")
            return False
```
